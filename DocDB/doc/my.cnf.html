/etc/my.cnf

Edit /etc/my.cnf to reflect the location and owner of your MySQL database. The file should look something like this: 

[mysqld]
user=mysql_owner
datadir=/path/to/datadir/mysql
socket=/path/to/datadir/mysql/mysql.sock
skip-innodb
     
[mysql.server]
user=mysql_owner
basedir=/path/to/datadir

[client]
user=mysql_owner
socket=/path/to/datadir/mysql/mysql.sock

[safe_mysqld]
err-log=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid
